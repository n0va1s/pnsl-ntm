```php
it('arquivar equipe aplica soft-delete preservando pivot equipe_usuario (EQUIPE-10)', function () {
    $this->actingAs($this->coordGeral);

    // Confirmar que pivot existe antes do arquivamento
    expect(
        \Illuminate\Support\Facades\DB::table('equipe_usuario')
            ->where('idt_equipe', $this->equipe->idt_equipe)
            ->count()
    )->toBe(2);  // coordGeral + membro

    Volt::test('equipes.index')
        ->call('arquivar', $this->equipe->idt_equipe)
        ->assertHasNoErrors();

    // Equipe não aparece em query padrão (soft-deleted)
    expect(Equipe::find($this->equipe->idt_equipe))->toBeNull();

    // Equipe existe com withTrashed e tem deleted_at
    $equipeSoftDeleted = Equipe::withTrashed()->find($this->equipe->idt_equipe);
    expect($equipeSoftDeleted)->not->toBeNull();
    expect($equipeSoftDeleted->deleted_at)->not->toBeNull();

    // Pivot PRESERVADA — cascadeOnDelete não dispara em soft-delete (apenas em forceDelete)
    expect(
        \Illuminate\Support\Facades\DB::table('equipe_usuario')
            ->where('idt_equipe', $this->equipe->idt_equipe)
            ->count()
    )->toBe(2);
});
```
